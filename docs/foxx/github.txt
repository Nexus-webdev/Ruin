# Git;

# The '-git-' object, can access github;
-git-:cd <owner>/<repo>/<subdirectory (optional)>;

<var> = -git-:read <path-to-file>;
-git-:write <path-to-file> >> <content>
-git-:rm <path-to-file>;

-git-:mkdir <path-to-dir>;
-git-:rmdir <path-to-dir>;

-git-:install <alias>: <owner>/<repo>/<path-to-dir>;
