# Popups;
popup:show 'A' >> 'Please Enter Your Name';
doc:listener #A-popup-ok: <click> {
 print getValueOf 'A';
};

doc:listener #A-popup-close: <click> {
 popup:show 'A' >> 'You Must Enter Your Name';
};

function getValueOf(id) {
 const res = doc:query `#${id}-popup-response`;
 return res.value;
}
